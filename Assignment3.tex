	\section{Question:-}
	\paragraph{In $\Delta$ABC, given that a+b+c = 11, $\angle$B = 45$^{\circ}$ 
		and
		$\angle$C = 45$^{\circ}$ 
		, find a, b, c and sketch the triangle.}
	\section{Solution:-}
	\begin{tikzpicture}[scale=2]
		
		%Triangle sides
		\def\a{4}
		\def\c{3}
		
		%Marking coordiantes
		\coordinate [label=above:$B$] (B) at (0,\c);

		%Drawing triangle ABC
		\draw (B) -- node[left] {$\textrm{c}$} (A) -- node[below] {$\textrm{a}$} (C) -- node[above,,xshift=2mm] {$\textrm{b}$} (B);

		\tkzMarkAngle[fill=orange!40,size=0.5cm,mark=](B,C,A)

	\end{tikzpicture}
\newline
Given, 
\begin{equation}
	a+b+c=11
\end{equation}
\newline
We know that, 
\newline
tan(ACB)=$\frac{c}{a}$
\newline
tan(45$^{\circ}$)=$\frac{c}{a}$
\begin{equation}
	c=a
\end{equation}
\newline
Now, $a^2$+$c^2$=$b^2$   (Pythagoras Theorem)
\newline
Using eq 2) 
\begin{equation}
	$$2$a^2$=$b^2$$$
\end{equation}
Also, eq 1) becomes 
\begin{equation}
	b=11-2a    
\end{equation}
\newline
Therefore, eq 3) becomes
\newline
2$a^2$=$(11-2a)^2$
\newline
2$a^2$-44a+121=0
\newline
a=3.25,a$\neq$18.5
\newline
b=11-2*3.25=4.5
\newline
\paragraph{a=3.25 , c=3.25 , b=4.5}
